## **Паттерны работы с БД**

### **Single Flight**

> - у нас есть популярный пользователь или узел
> - приходит запрос на его получение: мы идем в кэш, затем в БД
> - таких запросов очень много, поэтому сразу много тяжелых запросов ложится в БД и кладет ее
> - чтобы этого избежать можно воспользоваться single flight – один запрос сходит в БД, а другие дождутся результата и вернут его
> - эта проблема называется **thundering herd**

```go
type call struct {
	err error
	value any
	done chan struct{}
}

type SingleFlight struct {
	mutex *sync.Mutex
	calls map[string]*call
}

func NewSingleFlight() &SingleFlight{
	calls: make(map[string]*call),
	mu:    &sync.Mutex{},
}

func (s *SingleFlight) Do(ctx context.Context, key string, action func (ctx context.Context) (any, error)) (any, error) {
	s.mutex.Lock()
	if call, found := s.calls[key]; found {
		s.mutex.Unlock()
		return s.wait(ctx, call)
	}
	
	call := &call{
		done: make(chan struct{}),
	}
	s.calls[key] = call
	s.mutex.Unlock()
	
	go func () {
		defer func() {
			if v := recover(); v != nil {
				call.err = errors.New("error from single flight")
			}
			close(call.done)
			
			s.mutex.Lock()
			delete(s.calls, key)
			s.mutex.Unlock()
		}()
		
		call.value, call.err = action(ctx)
	}()
	
	return s.wait(ctx, call)
}

func (s *SingleFlight) wait(ctx context.Context, call *call) (any, error) {
	select {
		case <-ctx.Done():
			return nil, ctx.Err()
		case <-call.done:
			return call.value, call.err
	}
}
```

### **Moving Later**

> - есть кластер БД
> - внутри кластера синхронная репликация
> - сервис хочет сделать распределенный запрос
> - используется **first response – win стратегия**
> - то есть есть распределенный кластер, опрашиваем несколько реплик, первая которая отвечает – побеждает, от остальных просто перестаем ждать ответ
> - да, грузим дополнительно кластер, но зато минимальное время ответа 

```go
type Database struct {
	Query(query string) string
}

func DistributedQuery(replicas []Database, query string) string {
	responseCh := make(chan string, 1) // обязательно буферизованный, иначе все горутины могут выйти в default-е
	
	for _, replica := range replicas {
		go func() {
			select {
				case responseCh <- replica.Query(query):
				default:
					return
			}
		}()
	}
	return <- responseCh
}
```

### **Error Group**

> запрос в распределенный кластер да еще и с несколькими шардами
> хотим пойти во все шарды, выполнить запрос, собрать все данные со всех шардов
> если где-то ошибка (в одном из шардов), то выполнять остальные нет смысла

**usage**
```go
func DistributedQuery(ctx context.Context, shards []Database, query string) ([]string, error) {
	mutex := &sync.Mutex{}
	responses := make([]string, 0, len(shards))

	group, ctx := NewErrGroup(ctx)
	
	for _, shard := range shards {
		group.Go(func() error {
			type result struct {
				response string
				err      error
			}
			
			resultCh := make(chan result, 1)
			go func() {
				response, err := shard.Query(query)
				resultCh <- result{response: response, err: err}
			}()
			
			select {
				case <-ctx.Done():
					return ctx.Err()
				case result := <- resultCh:
					if result.Err != nil {
						return result.Err
					}
					
					mutex.Lock()
					responses = append(responses, result.response)
					mutex.Unlock()
					
					return nil
			}
		})
	}
	
	if err := group.Wait(); err != nil {
		return nil, err
	} else {
		return responses, nil
	}
}
```

**implementation**
```go
type ErrGroup struct {
	cancel func(error) // this is antipattern, but ok here...
	wg     sync.WaitGroup

	errOnce sync.Once
	err     error
}

func NewErrGroup() *ErrGroup {
	ctx, cancel := context.WithCancelCause(ctx)
	return &ErrGroup{
		cancel: cancel,
	}, ctx
}

func (g *ErrGroup) Go(action func() error) {
	g.wg.Add(1)
	go func() {
		defer g.wg.Done()
		if err := action(); err != nil {
			g.errOnce.Do(func() {
				g.err = err
				g.cancel(g.err)
			})
		}
	}()
}

func (g *ErrGroup) Wait() error {
	g.wg.Wait()
	return g.err 
}
```

## **Best practices fault tolerance и resilience**

### **Кратко**

> **Graceful shutdown** в Go реализуется через обработку SIGTERM и вызов `http.Server.Shutdown`, который завершает активные запросы перед остановкой процесса. Важно сначала дождаться завершения запросов, и только потом закрывать ресурсы вроде базы данных.
> 
> **Connection draining** — это отдельный механизм на уровне Kubernetes: через readinessProbe pod исключается из балансировки, и новые запросы в него больше не приходят, хотя существующие соединения продолжают обрабатываться.
> 
>Вместе они обеспечивают **zero-downtime**: draining останавливает входящий трафик, а shutdown корректно завершает текущие запросы.

### **Graceful shutdown**

> Корректное завершение работы **приложения как процесса**: перестаём принимать новые запросы и аккуратно завершаем текущие.
> **Суть:** аккуратно выключаем приложение.
#### **Как это работает**
```
- получаем `SIGTERM`
- закрываем listener (не принимаем новые соединения)
- ждём завершения активных запросов / горутин
- освобождаем ресурсы (DB, очереди)
- выходим из процесса
```
#### **Пример на Go**
```go
srv := &http.Server{Addr: ":8080"}

go func() {
    <-ctx.Done()

    shutdownCtx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
    defer cancel()

    _ = srv.Shutdown(shutdownCtx)
}()

_ = srv.ListenAndServe()
```

### **Graceful degradation**

> Стратегия, при которой система **продолжает работать при сбоях или перегрузке, но с ухудшением функционала**.
> **Суть:** система деградирует, но не падает.
#### **Как это работает**
```
- при ошибках или таймаутах:
    - используем fallback (кэш)
    - отключаем тяжёлые фичи
    - уменьшаем качество ответа
- часто используется:
    - timeouts
    - circuit breaker
    - retries
```
#### **Пример на Go**
```go
ctx, cancel := context.WithTimeout(ctx, 100*time.Millisecond)
defer cancel()

data, err := slowService.Get(ctx)
if err != nil {
    return cache.Get()
}
return data
```

### **Graceful connection draining**

> Механизм **корректного вывода инстанса из трафика через readinessProbe**, чтобы новые запросы не приходили, но старые завершились.
> **Суть:** аккуратно выводим инстанс из трафика, не роняя активные запросы.
#### **Как это работает**
```
1. pod помечается `not ready` (readinessProbe = false)
2. load balancer перестаёт слать новые запросы
3. старые соединения продолжают работать
4. после draining timeout (время для обновления маршрутизации) → shutdown процесса
```
#### **Отличие от graceful shutdown**
```
[ readiness=false ] → connection draining (инфраструктура)  
[ SIGTERM ] → graceful shutdown (приложение)
```

### **Readiness vs Liveness probes**
- **readinessProbe** → влияет на трафик, который идет на под (draining)
- **livenessProbe** → не относится к draining, просто перезапускает pod

### **Жизненный цикл остановки pod в Kubernetes**

Когда pod убивают (rolling update, scale down):
1. Kubernetes отправляет **SIGTERM**
2. pod:
    - **должен перестать принимать новый трафик**
    - **должен корректно завершить текущие запросы**
3. ждёт `terminationGracePeriodSeconds` (по умолчанию 30s)
4. если не успел → **SIGKILL**
#### **Пример на Go
```go
srv := &http.Server{Addr: ":8080"}  
  
go func() {  
	if err := srv.ListenAndServe(); err != nil && err != http.ErrServerClosed {  
		log.Fatalf("listen: %s\n", err)  
	}
}()
  
signalChan := make(chan os.Signal, 1)  
signal.Notify(signalChan, syscall.SIGINT, syscall.SIGTERM)  
  
// ждем сигнала от ОС / kubernetes о том, что нужно завершать работу
<-signalChan

shutdownCtx, cancel := context.WithTimeout(context.Background(), 10*time.Second)  
defer cancel()
  
// shutdown даёт завершиться активным HTTP-запросам
if err := srv.Shutdown(shutdownCtx); err != nil {
	log.Println("shutdown error:", err)
}  
  
// теперь уже закрываем DB, останавливаем воркеры, чистим другие ресурсы; потому что если закроем до shutdown, то активные запросы могут еще работать с БД и получим ошибки
db.Close()
```

### **Сделай system design: “100k RPS сервис с zero-downtime deploy”**

> “zero downtime” = **нет ошибок для пользователя + нет заметного деградационного скачка latency**
#### **High-level архитектура**
```
Clients
  ↓
CDN / Edge
  ↓
L7 Load Balancer
  ↓
Kubernetes Service
  ↓
Pods (stateless Go app)
  ↓
Cache (Redis)
  ↓
DB (sharded / replicated)
```
#### **Идеальный production-grade shutdown шаблон для Go с errgroup, контекстами и воркерами**
```go
package main

import (
	"context"
	"errors"
	"log"
	"net/http"
	"os/signal"
	"sync/atomic"
	"syscall"
	"time"

	"golang.org/x/sync/errgroup"
)

func main() {
	// 1. Контекст с сигналами
	ctx, stop := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
	defer stop()

	g, ctx := errgroup.WithContext(ctx)

	// 2. Readiness флаг (для draining)
	var isReady atomic.Bool
	isReady.Store(true)

	// 3. Ресурсы
	db := NewDB()

	// 4. HTTP mux
	mux := http.NewServeMux()

	// --- бизнес endpoint
	mux.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
		if !isReady.Load() {
			http.Error(w, "shutting down", http.StatusServiceUnavailable)
			return
		}

		time.Sleep(100 * time.Millisecond) // имитация работы
		w.Write([]byte("ok"))
	})

	// --- readiness (используется Kubernetes)
	mux.HandleFunc("/ready", func(w http.ResponseWriter, r *http.Request) {
		if !isReady.Load() {
			http.Error(w, "not ready", http.StatusServiceUnavailable)
			return
		}
		w.WriteHeader(http.StatusOK)
	})

	// --- liveness (процесс жив)
	mux.HandleFunc("/live", func(w http.ResponseWriter, r *http.Request) {
		w.WriteHeader(http.StatusOK)
	})

	srv := &http.Server{
		Addr:    ":8080",
		Handler: mux,
	}

	// 5. HTTP сервер
	g.Go(func() error {
		log.Println("http server started")
		if err := srv.ListenAndServe(); err != nil && !errors.Is(err, http.ErrServerClosed) {
			return err
		}
		return nil
	})

	// 6. Воркеры
	g.Go(func() error {
		return worker(ctx, "worker-1")
	})

	g.Go(func() error {
		return worker(ctx, "worker-2")
	})

	// 7. Shutdown + draining
	g.Go(func() error {
		<-ctx.Done()

		log.Println("shutdown signal received")

		// 7.1 выключаем readiness → начинаем draining
		isReady.Store(false)
		log.Println("readiness = false (start draining)")

		// 7.2 даём время LB/Kubernetes перестать слать трафик
		time.Sleep(5 * time.Second)

		// 7.3 graceful shutdown HTTP
		shutdownCtx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
		defer cancel()

		if err := srv.Shutdown(shutdownCtx); err != nil {
			log.Println("http shutdown error:", err)
		}

		// 7.4 закрываем ресурсы
		log.Println("closing db")
		db.Close()

		return nil
	})

	// 8. Ждём завершения
	if err := g.Wait(); err != nil {
		log.Println("exit with error:", err)
	} else {
		log.Println("graceful shutdown complete")
	}
}

// --- worker
func worker(ctx context.Context, name string) error {
	log.Println(name, "started")

	ticker := time.NewTicker(1 * time.Second)
	defer ticker.Stop()

	for {
		select {
		case <-ctx.Done():
			log.Println(name, "stopping")
			return nil
		case <-ticker.C:
			log.Println(name, "working")
		}
	}
}

// --- DB mock
type DB struct{}

func NewDB() *DB { return &DB{} }

func (db *DB) Close() error {
	log.Println("db closed")
	return nil
}
```
